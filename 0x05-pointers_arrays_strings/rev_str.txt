#include "main.h"
/**
 * rev_string - reverses a string
 * @s: string to be reversed
 *
 * Return: returns nothing
 */
void rev_string(char *s)
{
	int i = 0;
	int j;
	int k;
	int l;

	while (1)
	{
		if (s[i])
		{
			i++;
		}
		else if (!s[i])
		{
			break;
		}
	}
	k = i - 1;
	if (1)
	{
		char holder_str[i];

		for (j = 0; j < i; j++)
		{
			*(holder_str + j) = s[k];
			k--;
		}
		for (l = 0; l < i; l++)
		{
			*(s + l) = holder_str[l];
		}
	}
}
